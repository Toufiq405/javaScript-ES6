Extends

class TeamMembers {

  name;

  location;

  constructor(name, location) {

    this.name = name;

    this.location = location;

  };

  proviedFeedback() {

    console.log(`${this.name} Thank you for your feedback.`);

  }

};

class Instructor extends TeamMembers {

  designation = 'Web Course Instructor.';

  team = 'web team';

  constructor(name, location) {

    super(name, location);

  }

  startSupportSession(time) {

    console.log(`Start the support session at ${time}`);

  }

  createQuiz(module) {

    console.log(`Please create quiz for module ${module}`);

  }

};

class JobPlacement extends TeamMembers {

  designation = 'Job Placement';

  team = 'jobPlacement';

  resion;

  constructor(name, location,resion) {

    super(name, location);

    this.resion = resion;

  }

  startSupportSession(time) {

    console.log(`Start the support session at ${time}`);

  }

  createQuiz(module) {

    console.log(`Please create a quiz for module ${module}`);

  }

};

How to add Spinner to Website

const toggleSpinner = (isLoading) => {

  const spinnerContainer = document.getElementById('loader');

  if (isLoading) {

    spinnerContainer.classList.remove('d-none');

  } else {

    spinnerContainer.classList.add('d-none');

  }

}

// Start Loading or spinner

  toggleSpinner(true);

// Stop Loading or spinner

  toggleSpinner(false);
